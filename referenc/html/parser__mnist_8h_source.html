32 /*
33  *常用的CPU中的Intel处理器用小端格式，因此需要把大端数据
34  *转换为小端是数据进行处理。转换操作为将低8位（0~8位）左移24位，
35  *变成了高8位（24~32位），8~16位左移8位变成了（16~24位）。
36  *将原高8位和高16位右移，变成了新的低8位和低16位。
37 */
38 inline uint32_t Bige2Little (uint32_t value)
39 {
40  return (value & 0x000000FFU) << 24 | (value & 0x0000FF00U) << 8 |
41  (value & 0x00FF0000U) >> 8 | (value & 0xFF000000U) >> 24;
42 }
